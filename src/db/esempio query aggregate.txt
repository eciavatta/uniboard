db.classi.aggregate([ 
	{ $match: { nome: "aula1" } },
	{ 
		$lookup: { 
			from: "orari",
			let: {"classe": "$_id"},
			pipeline: [{
				$match: { 
					$expr: {
						$and: [
							{ $eq: ['$classe', '$$classe'] },
							{ $eq: ['$data', new Date("2019-07-04")] }
						]
					}
				}
			}],
			as: "orario"
		}
	}
])
